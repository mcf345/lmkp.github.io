    # 一、Vuex基础结构

    Vuex中state的值是不能直接进行修改的。
这是由于 Vuex 中的 state 引入了响应式数据的概念，Vuex 在 state 中使用了 Vue 的响应式机制。
直接修改state中的值，不会触发响应式机制，导致不会触发更新视图的操作。


修改state值的主要方法：
1)、使用 commit：通过在组件中调用 this.$store.commit('mutationName') 函数，触发 mutation 中的函数，从而修改 state 中的值。
2)、使用 dispatch：通过在组件中调用 this.$store.dispatch('actionName') 函数，触发 action 中的函数，从而间接修改 state 中的值。
3)、使用 mapMutations 和 mapActions：通过在组件的 computed 或 methods 中映射 mutation 和 action 中的方法，从而修改 state 中的值。

    # 1、结构

    import Vue from 'vue'
import Vuex from 'vuex'
// 模块化


import MachFeModule from './module/MachFe.js'


Vue.use(Vuex)


const store = new Vuex.Store({
 state: {
  data:'',
 },
 mutations: {
        /**
   * value.Name1 - 要赋值的属性名
   * value.Name2 - 赋值属性 属于对象下的属性，此时 Name1 就是对象名
   * value.data  - 赋值内容
   * value.StorageName  - 缓存名称，有缓存时会开启缓存
   * value.StorageData  - 缓存内容，已处理成JSON格式
   * @param {object} value  这是一个对象，包含属性名，赋值内容，缓存名、缓存内容
   */
  ChangeAndSetStateData(state, value) { // 修改 state 中的内容，有缓存就开启缓存
   if (value.Name1 && value.Name2) {
    state[value.Name1][value.Name2] = value.data;
   } else if (value.Name1 && !value.Name2) {
    state[value.Name1] = value.data;
   }


   if (value.StorageName && value.StorageData) {
    uni.setStorageSync(value.StorageName, value.StorageData)
   }
  },
 },
 actions: {
  receiceCode({
   commit
  }, value) {
   commit('RECEIVECODE', value)
  }
 },
 getters: {
  data: state => state.data,


 },
 modules: {
  MachFeVuex: MachFeModule // 将主题模块添加为名为"MachFeVuex"的模块
  // 可以添加更多的模块
 },
})


export default store

// 获取模块中的数据
this.$store.state.模块名.user


// 使用getter获取数据
this.$store.getters.HistoryDevice

// 调用index中的

    Name1: 'state变量',
